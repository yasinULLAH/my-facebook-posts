د غفلت خطرونه او د الله ذکر

إِنَّ الْحَمْدَ لِلَّهِ نَحْمَدُهُ وَنَسْتَعِينُهُ وَنَسْتَغْفِرُهُ وَنَعُوذُ بِاللَّهِ مِنْ شُرُورِ أَنْفُسِنَا وَمِنْ سَيِّئَاتِ أَعْمَالِنَا مَنْ يَهْدِهِ اللَّهُ فَلَا مُضِلَّ لَهُ وَمَنْ يُضْلِلْ فَلَا هَادِيَ لَهُ وَأَشْهَدُ أَنْ لَا إِلَهَ إِلَّا اللَّهُ وَحْدَهُ لَا شَرِيكَ لَهُ وَأَشْهَدُ أَنَّ سَيِّدَنَا وَمَوْلَانَا مُحَمَّدًا عَبْدُهُ وَرَسُولُهُ۔ أَمَّا بَعْدُ فَأَعُوذُ بِاللَّهِ مِنَ الشَّيْطَانِ الرَّجِيمِ۔ بِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ۔

وَالَّذِينَ اتَّقَوْا إِذَا مَسَّهُمْ طَائِفٌ مِّنَ الشَّيْطَانِ تَذَكَّرُوا فَإِذَا هُم مُّبْصِرُونَ * وَإِخْوَانُهُمْ يَمُدُّونَهُمْ فِي الْغَيِّ ثُمَّ لَا يُقْصِرُونَ

وَقَالَ تَعَالَىٰ

قَدْ أَفْلَحَ مَن تَزَكَّىٰ * وَذَكَرَ اسْمَ رَبِّهِ فَصَلَّىٰ * بَلْ تُؤْثِرُونَ الْحَيَاةَ الدُّنْيَا * وَالْآخِرَةُ خَيْرٌ وَأَبْقَىٰ * إِنَّ هَٰذَا لَفِي الصُّحُفِ الْأُولَىٰ * صُحُفِ إِبْرَاهِيمَ وَمُوسَىٰ

صَدَقَ اللَّهُ الْعَظِيمُ وَصَدَقَ رَسُولُهُ النَّبِيُّ الْكَرِيمُ وَنَحْنُ عَلَى ذَلِكَ مِنَ الشَّاهِدِينَ وَالشَّاكِرِينَ وَالْحَمْدُ لِلَّهِ رَبِّ الْعَالَمِينَ۔

درود شریف پڑھیے

د مسلمان غفلت

جماعت ته حاضرینو مسلمانانو وروڼو، مسلمان د غفلت نه وي، او د الله جل جلاله نه غافل نه وي، د غفلت په خوب ويده نه وي. مسلمان بايد ويښ وي. په دې سورت کښې چې کوم د شیطان برعکس ذکر راغلی دی، نو غفلت ورته نه دی، چې انسان پرې غالب وي او بيداره وي. او بیا چې کله غفلت راغلی وي، نو وعده صلاة وه، چې د منبر نه خير، د وعظ په ذريعه، د قران کریم د آیت، د رسول الله صلی الله عليه وسلم د حدیث مبارک، امر بالمعروف، نهی عن المنکر، د وعظ وعظ، او نصیحت کوم نصیحت، د بیان کوم بیان، هغه نه پاتې شي. غافل نه کړی شي. دا دومره خطرناک وخت دی، دا الله جل جلاله قهر، عذاب، د غفلت په دوران کښې راځي. الله دې مونږ د خپل قهر، غضب او عذاب نه وژغوري.

د الله عذاب او د غفلت وخت

څومره قومونه چې په دنیا کښې هلاک شوي دي، په هغې باندې چې کله د الله جل جلاله د طرف نه عذاب راغلی دی، قران کریم ته تاسو وګورئ، په هغې کښې به غافلون وو، غافلین د هلاکت راتلل، د هلاکت په ذریعه. دا خلق په غفلت کښې وو. یا دا خلق د څوک په حالت کښې وو، یا الله پاک فرمایي "ضحی" وهم یلعبون، "دوی د چاښت په وخت لوبې کولې". دوی د ساعت په وخت لوبې کولې. او د ساعت په وخت لوبې کول نو مراد د دنیا په کاروبار کښې مشغولي ده. لکه چې کله غرمه شي، دغه وخت کښې آخر خلق پاتې شي چې کوم د ناوخته ويده وي. زمیندار، باغوان په پټي کښې په باغ کښې لګیا وي، دکاندار به په دکان کښې لګیا وي، تاجر به تجارت کښې لګیا وي، د فیکټرۍ مالک او مزدور به په فیکټرۍ او کارخانه کښې لګیا وي. د دفتر ملازم به په دفتر خلاص کړی په هغه کښې ناست وي. دې ته قران کریم د لوبو حالت "وهم یلعبون"، "په دې وخت کښې وو چې د الله عذاب راغی، چې دوی لوبې کولې". لوبې کښې مشغول وو. د لوبو مختلف قسمونه دي. چې ماشوم، ماشوم په څو ورځو لوبې کوي؟ بیخي چې کله لږ شي، تګړی شي، نو هغه لوبې کوي. بیخي چې کله لږ غټ شي، د سر، د کور، د وجود، د ښځې، د کار و بار غم پیدا شي، بیا د ده لوبې جلا دي. بیا د ده لوبې جلا دي. چې څنګه د ماشومتوب لوبې پسې د ځوانۍ کښې سړی خندا ټکې مسخرې کوي. دا درسره ستا دا حرکات چې په دې کښې د الله جل جلاله یاد نشته دا غفلت دی. چې سړی په دنیا کښې لګیا وي، په مارکیټ کښې لګیا وي، په کاروبار کښې لګیا وي، په لوړ منصب کښې لګیا وي مشغول وي، او د الله جل جلاله نه دې یاد وي. دا دې په لوبو کښې پاتې وي.

نو عرض مې دا کولو چې د الله جل جلاله عذاب هغه وخت راځي چې کله انسان په غفلت کښې وي. مسلمان غفلت کښې نه وي. دی د الله جل جلاله یادوي.

د متقي او غافل انسان فرق

قران کریم په دې آیت کښې الله جل جلاله دا وفرمایل وَالَّذِينَ اتَّقَوْا إِذَا مَسَّهُمْ طَائِفٌ مِّنَ الشَّيْطَانِ تَذَكَّرُوا چې څوک متقیان وي، پرهیزګاره وي، تقوی دار وي یعنې مومنان د الله نه یریږي، د زړه په تلی کښې د الله جل جلاله خوف او ویره شته، إِذَا مَسَّهُمْ طَائِفٌ مِّنَ الشَّيْطَانِ چې کله د شیطان د طرف نه یو وسوسه دوی ته راشي، په زړه کښې وسوسه راشي، دا ګناه وکړه، دا ګناه وکړه، دا ګناه وکړه. هغه د حرام مال د غلا سوچ شي. د مال واخله، دغه غلا کړه، دغه شے غصب کړه، دا امانت دې، د مالک ته مه ورکوه، دا امانت پټ کړه، بل باندې منکر شه. دا شیطان کوي. نو تَذَكَّرُوا دغه وخت دوی ته الله جل جلاله یاد شي. فَإِذَا هُم مُّبْصِرُونَ "نو په دې وخت د دوی سترګې خلاصې شي". "دوی لیدل پیل کوي". دوی بیا بصیرت الله جل جلاله پیدا کړي، او د دې نه زړه ساتی چې دا خو حرام دي، دا خو سمه نه ده. الله جل جلاله یاد کړي.

رسول الله صلی الله علیه وسلم په یو حدیث مبارک کښې د تیرو امتونو د دوه کسانو واقعه ذکر کړې. زمونږ په طرف کښې اوله تشبیحي پښتنو کښې وي، د دنیا د چا نه مخه اباده ده. او دا په حقیقت کښې د حدیث مبارک مفهوم دی. د دنیا د چا نه مخه اباده ده. یعنې په دې دنیا کښې نیکان خلک شته، د الله جل جلاله یاد په بعضې خلکو په زړونو کښې شته او دا دنیا تر هغې اباده ده، کوټه اباده ده، مکان تر هغې پورې اباد دی چې په هغې کښې اوسیدونکی او مکین موجود وي. چې کله اوسیدونکی ترې ختم شي، بیا د مکان پلستر، دیوالونه، چت، دا هر څه وران شي. نو دا د چا نه مخه اباده ده، دا هغه کس دی چې څوک الله یادوي. او حدیث مبارک کښې راځي، لَا تَقُومُ السَّاعَةُ حَتَّى لَا يُقَالَ فِي الْأَرْضِ اللَّهُ اللَّهُ، "تر هغې پورې قیامت قایم نه شي، تر څو په ځمکه کښې د الله نوم اخیستونکی وي". الله الله ویونکی وي. مسلم شریف کښې حدیث دی. تر هغې پورې به قیامت نه راځي تر څو پورې چې الله الله ویونکی وي. د الله نوم اوچتوونکی وي.

نو چې کله تقوی دار، پرهیزګار، مومن، مسلمان سړی وي، د ده ته چې شیطان راشي او د ده په زړه کښې یو غلط وسوسه واچوي، هغه فوراً الله جل جلاله یادوي. او، دا خو زه په ګمراهۍ کښې وم، بس هغه د الله جل جلاله نوم ووايي، او د شیطان د دې وسوسې، اندیښنې، سودا نه بچ شي.

د امانت او تقویٰ یوه واقعه

رسول الله صلی الله علیه وسلم واقعه بیان کړه، بخاري شریف کښې حدیث دی، نورو کتابونو کښې هم شته. یو سړي په بل باندې ځمکه خرڅه کړه. په ځمکه کښې کرڅوڼي ته باړۍ ووهي یا د شنو ته مستلي. بس په کر کښې یې ځمکه واخیسته. هغه چې کله ځمکه یې وکرله، چې کله یې وکرله، نو په هغې کښې یې وکتل، نو هغه زوړ حدیث مبارک کښې مختصر واقعه بیان شوې ده، تاسو ته لږونه کله په کاله کښې تشریح کوم. هغه چې یې وکتل نو هغه یو کوږوی راووت، ډک ډک دی، په هغې کښې خزانه ده. قیمتي اشیاء پکښې پرتې دي. زر دي پکښې، یا لعلونه دي، جواهرات دي. چې دا یې وکتل، فوراً بیرته وویل چې دا خو ما چې ځمکه اخیستې ده، دا د هغه مال دی. ځکه چې ما خو ځمکه اخیستې وه، په ځمکه کښې زه ایوبي کوم، نو چې چې څه راووتل په هغې کښې دا د هغه خلکو دی. بیا یې نه ونه، چې څوک سره به مال ډیر شو، دا به یې ځمکې کښې کټوۍ کښې ښخ کړو. یا به یې دېوال په لاندې کښې کړو.

قران کریم وګورئ، سوره کهف کښې الله جل جلاله ذکر کړی دی چې کله موسی علیه السلام او خضر علیه السلام په کښتۍ کښې تلل، دیوال یې جوړ کړو. آخر خضر علیه السلام ته موسی علیه السلام وویل چې دا ستا دیوال نه جوړوم. موسی علیه السلام ورته وویل دا څه یې؟ خلق په مونږه میلمستیا نه کړه، مونږ ته یې روټۍ ونه کړه، ته لګیا یې دوی له ته سیمټ اچوې؟ نو خضر علیه السلام بیا وروسته موسی علیه السلام ته چې د هغه درې واړو واقعاتو حقیقت بیان کړو، نو دیوال په اړه یې دا وفرمایل چې دا د دوه یتیمانو دی. او د هغوی پلار مړ شوی وو. وَكَانَ أَبُوهُمَا صَالِحًا، "او د هغوی پلار نیک سړی وو". هغه مړ شوی وو. هغه خزانه یې جوړه په دیوال کښې کړې وه. دیوال چې څه وخت راغورځیدلی، یو یې د دې لور ته سر، د بل د هغه لور نه. چې دا دیوال غورځیدلی وی، هغه به دا خزانه راښکاره شوې وه، نو بیا به یې خلکو ترې چور کړی وه، مال به یې ترې خوړلی وو. نو دا دیوال مې سم کړو چې دا مال پټ شي.

نو دا سړي فکر وکړو، دا زما نه دی، دا خزانه د هغه ده. دا یې محفوظ کړی وو، سړی بېک وو. هغه یې زر راواخیستو. او مالک ته یې راوړو. مالک ته یې وویل چې تا نه ما ځمکه اخیستې وه نو په هغې کښې داسې داسې مال راووتلی، دا خپل مال واخله. دا هغه دی، وَالَّذِينَ اتَّقَوْا إِذَا مَسَّهُمْ طَائِفٌ مِّنَ الشَّيْطَانِ تَذَكَّرُوا، چې کله شیطان په زړه کښې وسوسه واچوي، فوراً الله جل جلاله یاد کړي. د شیطان د ګمراه کولو نه مخکښې مخکښې مومن هغه د شیطان د وسوسې نه ځان وژغوري. شیطان خو دغه وخت سړي ته وايي، پټ یې کړه. چا لیدلی؟ دا مال پټ کړه، ځان سره یې محفوظ کړه. هغه ورور یې، چې کله یې راوړو نو د هغه بل ورور هم دا ډول مومن، مسلمان، پاک، متقي او پرهیزګاره انسان وو. هغه ورته وویل چې دا خو ما نه دي خرڅ کړی. زه ورته وویل چې تا نه مې ځمکه خرڅه کړې وه، ما تا باندې ځمکه خرڅه کړې ده، سره د هغې څه. ما تا باندې ځمکه خرڅه کړې ده، سره د هغې څه شته، دا ستا دي.

د دې دواړو راغلی اختلاف. اختلاف په دې شو چې هغه یو وايي دا مال ستا دی، بل وايي ستا دی. وَالَّذِينَ اتَّقَوْا إِذَا مَسَّهُمْ طَائِفٌ مِّنَ الشَّيْطَانِ تَذَكَّرُوا، "هغه خلک چې مومن دي، متقي دي، پاک او پرهیزګاره دي، کله چې دوی ته د شیطان له لوري یو وسوسه راشي، نو دوی الله یادوي". دوی دواړو یو بل نه، بل د نماز لګیو. دا مسئله یې یو بل کس ته راوړه. هغه یو قاضي سړی وو، یا پیغمبر وقت وو، مطلب چې داسې کیسه ده. هغه ته یې پوښتنه وکړه، دواړو ته یې وویل چې ستا زوی شته که لور؟ بل ته یې وویل ستا زوی شته که لور؟ د یو زوی د بل لور، د هغوی په منځ کښې یې واده وکړو، دا مال یې په هغوی خرڅ کړو. جهیز په سامان باندې، په مهر باندې، جوړجاړی راغی او یو بل ته یې بخشش کړل. دا مسئله هواره شوه. هغه عرض مې دا کولو چې مومن غافل نه وي. مومن غافل نه وي. هغه غفلت کښې نه وي. هغه چې کوم ځای کښې شیطان دوکه کړي، هلته هغه الله جل جلاله یاد کړي.

د عبدالله بن مسلمه القعنبي د توبې واقعه

څومره د غفلت تربیه، هغه د الله د ذکر سره. د الله یاد سره، چې انسان د زړه نه لري وي، او چې څومره سړی د الله جل جلاله د یاد نه لري وي، نو بیا یې نه یادیږي. د الله نوم نه اخلي، نو بیا به په زړه کښې لړه پسې جمع کیږي جمع کیږي جمع کیږي، بیا په زړه کښې نه خیر نه ننوځي، نه هدایت ننوځي.

لَهُمْ قُلُوبٌ لَّا يَفْقَهُونَ بِهَا "الله جل جلاله فرمایي د هغوی زړونه شته، خو په هغو زړونو باندې پوهه نه کوي". د سوچ کار نه اخلي. فَلَمَّا زَاغُوا أَزَاغَ اللَّهُ قُلُوبَهُمْ، بني اسرائیلو په اړه کښې الله جل جلاله فرمایي، حضرت موسی علیه السلام قوم امت، چې کله هغوی کاږه شول، کږه شول، أَزَاغَ اللَّهُ قُلُوبَهُمْ نو الله جل جلاله د هغوی زړونه کاږه کړل. او چې کله زړه کوږ شي، زړه زنګ شي، زړه سخت شي، زړه تور شي، بیا په هغې کښې خیر نه ننوځي. بیا یې واعظونه، نصیحتونه، بیانونه څه نه مني. هغې باندې بیا پرده لوځیږي او پردو کښې راګیر شي.

نو عرض مې دا کولو چې مونږ ته د الله جل جلاله یاد او ذکر، دا ډیر ضروري دی. دا د ګناه کښې د مبتلا کیدو نه مخکښې، یا د ګناه دوران کښې. د ګناه کښې د مبتلا کیدو نه مخکښې، حضرت یوسف علیه السلام د عزیز مصر ښځې په خپل محل کښې راوړو، دروازې یې بندې کړې، دروازې یې بندې کړې، داسې ځای ته یې بوتلو چې هلته نه د راوتلو د عقل او د حواسو په موجب باندې راوتل، هغه انتهائي ګران وو. لیکن سره د دې، حضرت یوسف علیه السلام منډه کړه، په شړنده ورپسې، منډه یې کړه، الله جل جلاله یې خلاص کړو، او الله پاک حضرت یوسف علیه السلام حفاظت وکړو. تذکر، معصیت نه مخکښې.

زړه کښې شیطان د وسوسې اچولو کوشش کړی، چې دلته راغلی، ته معصوم یې، الله جل جلاله دې وژغوره. کله داسې وي چې ګناه کړې وي، په ګناه کښې روان وي، په ګناه کښې بوخت وي، بیا الله جل جلاله ورته توفیق ورکړي، او بیا الله جل جلاله ته یاد شي. لکه څنګه چې سړی په غفلت خوب باندې ویده وي او څوک یې بیداره کړي، چې کله بیدار شي بیا الله جل جلاله ته ورشي، الله جل جلاله ورته د توبې توفیق ورکړي.

ابو داود شریف مطالعه کړی، دا حدیث دی. په هغې کښې یو محدث دی، راوي دی، د هغه نوم راځي، عبدالله بن مسلمه. عبدالله بن مسلمه، امام بخاري رحمه الله سره یې بخاري کښې هم د ده نه ډیر زیات احادیث روایت کړي دي. او د هغه په القعنبي نسبت باندې هم مشهور دی. امام مالک رحمه الله خاص شاګرد دی. د نورو محدثینو تلمذ او شاګردي یې هم اختیار کړې ده. ده د احادیثو د زده کولو نه مخکښې، او د توبه، د توبې د سوچ نه مخکښې، په خپل کښې هر یو غلط لاره څارلی وو. په خپل علاقه کښې غټ غونډه او بدمعاش وو. داسې بدمعاش وو چې څوک یې مخې ته نه راتلل. چا نه مال غلا کړی، د چا یې عزت بې عزتي کړې، چا سره یې څه کړي، چا سره یې څه کړي، چا ته یې د پلار نوم نه اخیستو.

یوه ورځ د یو مشهور محدث دی، محدثین، هغه ته یې یو لقب ورکړی دی، امیر المومنین فی الحدیث، شعبه بن الحجاج رحمه الله. دا محدث د دې درس مشهور وو. زرګونو د شاګردانو د ده په درس کښې ناست وو. ده په خپل درس ته په آس باندې راغلی وو. په هغه زمانه کښې غټه اعلی سواری وه. شعبه بن الحجاج رحمه الله، دېده کله راغلی وو، عبدالله بن مسلمه قعنبي، دې لار ته کتل، کتل نو داسې وویل، چې دا انتہائی بدکاره سړی په اړه، بیا ده ورته وویل چې ته زما په درس کښې، ته ناسته ونه، ما ته د ساعت مخکښې راشه، چې کله سړی په بدو ګناهونو کښې بوخت شي، د هغه په زړه باندې پرده وغورځیږي، هغه بیا د ساعت مخکښې نه شي.

یوه ورځ، ده ته یې وویل، په دې کښې شعبه رحمه الله درس ته چټي کړه، او په آس باندې سور شو، کور ته روان شو. شاګردان څه مخکښې، څه وروسته، څه یو په بل باندې، د دین، علوم، احادیثو کتابونه. په دې کښې د استاد او شاګرد کوم تعلق وي، د مینې، د محبت، بل طرف نه شفقت، مهرباني، هغه نه صرف په درس کښې وي، بلکې په درس نه بهر هم وي، په تګ او راتګ کښې وي، په سفر او حضر کښې وي، په مدرسې کښې وي.

شاګردان چې کله په کور ورغلي، په دې کښې مخامخ دې راغلی، عبدالله بن مسلمه قعنبي. دې چې کله د وارې را روان وو نو په غصه لګیا وو. دې راغلو، نو د شعبه رحمه الله مخ ته یې د لجام نه ونیو. او داسې په تکبر انداز سره، نږدې شو او ورته یې وویل چې هغه خلکو ته احادیث بیانوه، ما ته حدیث ووایه. شعبه رحمه الله ورته وویل چې د حدیث د اوریدلو او ویلو ځای نه دی. تره څه، ته راشه، جماعت ته، هلته درس کښې کینه، په درس کښې د نورو طالبانو کښې، نو هلته به یې کښېنه. دا داسې ورته وویل، د حدیث یې ورته ونه ویل او روان شو. لکه چې دا ماحول، دا سړی غلط ماحول کښې دی، په دې باندې غلط विचारو پیداشي. او ښه ماحول کښې سړی بیا د هغې اثر پرې پریوځي.

هلته اجتماع ده، خلک راغلي دي، په اباده، مسلمانانو په منځ کښې ناست دي، دا هم دې ته متوجه شي، او بیا شکرینه کړي. دومره په اخر کښې ثوابونه راشي. ماحول بدلیږي. ماحول کښې سړی تهجد ته پاڅیږي، نه کړې مسکي یې نه کړې. تلاوت د قران پاک کړې یا نه کړې، خو هلک لیدل کیږي. ذکر مخکښې کړی وي یا نه، چې ټول خلک ذکر کړي، دی هم لګیا شي. چالو شي. د رانیول، دا مجبوري کار دی، وران کار دی. دا د ماحول اثر دی.

د غفلت علاج: د الله ذکر

چې خراب ماحول کښې سړی وي، نو بیا د خولې نه کله کله کفریه کلمات راوځي. تبلیغي جماعت والو کښې ډیر داسې خلک دي چې اصول یې زده وي، پرهیزګاري ورسره وي، دې عمل کوي. هغه دا چې دوی چې کله د دعوت لپاره زي، ګشت کړي، دکاندارانو ته، یو ته بل ته دعوت ورکړي، نو هغوی ته وايي چې مونږ د دکان پریږده، راشه، زمونږ سره به خبرې وکړو. بس سیده مسجد ته راشه، هلته به بیا دینو خبرې وسو، بیا د ماحول بدل شي، د زړه بدل شي. بیا چې کله ورته دعوت ورکړي، بیا د هغوی نه پس ورسره بحث او سال ته نه کړي، که هغه سل جګړې کوي. ورته وايي چې ماته راشه. بیا چې جماعت ته راشي نو بیا ماحول ډیر پاک شي. بیا هلته خپله بد نشي، کله هم نشي. ځکه چې ټول ماحول پاک دی، زړه بدل شي. دا ډیر عجیب اصول دی. هغه ځای کښې چې کوم ځای د یو بدمعاش ته بدمعاشي وي، د یو غونډه په غونډګردي وي، د یو متکبر په تکبر وي، په هغه ځای کښې تا ورته په غندنه د قران ایت بیان کړې، او د رسول الله صلی الله علیه وسلم حدیث مبارک ورته بیان کړې، هغه سړی نه مني چې ایت او حدیث ته سپکاوی وشي. دا هغه سپکاوی دی.

او چې کله یې سپکاوی وکړو نو ته خو له هغې چې خیر ته رسې، دی بیا نور شر ته، کفر ته، اسلام نه خارج شي، د دین نه ووځي. ځکه چې د دین سپکاوی، قران پاک ته، د قران کریم ایت ته، د رسول الله صلی الله علیه وسلم ذات اقدس ته، د هغه حدیث مبارک ته سپکاوی کول، نو دا خو ظاهر دی چې د اسلام نه سړی ووځي.

نو شعبه بن الحجاج ورته وویل چې دا د حدیث بیانولو ځای نه دی. که حدیث ته اوریدل شوق لرې نو درس ته راشه. نو هغه یې د اس نه، د واګې یې ونیوله، او هغه یې ځان ته سړی بدمعاش کړو. ورته یې وویل چې بیان کړه، زه نه لېږدم. به پوه شې. خو څه خیال بدل شو. نو حدیث یې ورته بیان کړو. هغه حدیث مبارک په صحیح بخاري کښې هم شته. نبي کریم صلی الله علیه وسلم فرمایلي: إِنَّ مِمَّا أَدْرَكَ النَّاسُ مِنْ كَلَامِ النُّبُوَّةِ الْأُولَى إِذَا لَمْ تَسْتَحِ فَاصْنَعْ مَا شِئْتَ. "خلقو ته چې د تیرو انبیاء کرامو خبره مبارکه ورسیدلې ده، مشترکه خبره، هغه دا ده چې کله تا کښې حیا نه وي، او شرم درکي نه وي، چې څه دې خوښه وي هغه کوه." "کله چې تا کښې شرم او حیا نه وي، نو بیا څه دې خوښه وي هغه کوه". د ګناه کړه، بیا د طالب، ظالمان، برملا، په لېن کوڅ کښې، په بس کښې، په یو ځای کښې، په بل ځای کښې، په کلي کښې، په مجلس کښې، چې یو ته لګیا وي، بل ته لګیا وي، داسې ګناهونه کوي، په خلاص لګیا وي. ځکه چې حیا پکښې نه شته، سترګې نه شرمیږي.

دا حدیث یې ورته شعبه ووایه. نو عبدالله بن مسلمه باندې الله جل شانه دا واقعه فضل کړو، او هدایت ورته الله پاک ورکړو. الله جل شانه چې چا ته په غصه شي، دا یې نصیب کړي. تر څو پورې په ګناهونو کښې مبتلا وو، سمه یې کړه.

د توبې دروازه او د الله رحمت

ورانه دوکان ته مولا کسر ورکړو، سوچ یې کړو، چې دا نیک سړی دی، تقوی دار مسلمان دی، محدث دی، ما هغه نورو خلکو ته د غصه خپرولو له څه چول، خو اوس ما د دې ازار ورکولو تمه تکلیف ورکړو، دا د شياطينو ما ته شرم او حیا نه شته. په خلکو مې حیا په مخ کښې وه، شرم په مخ کښې وه، ما به دا سړی ووایه، چې زه ورته دا راغلی وم. بس د زړه کښې یو رڼا راغله. او هغه سره شعبه رحمه الله ته یې وویل، چې زه ستا په لاس توبه وباسم. ما د شعبه بن الحجاج په لاس باندې توبه وویسته. او عبدالله بن مسلمه رحمه الله دومره مشهور محدث جوړ شو، چې امام بخاري رحمه الله د هر کس نه، هر کس نه احادیث نه اخلي. هغه داسې راوي نه، او بیا د هغه سره سره، د امام بخاري د شرط مطابق وي.

نو عبدالله بن مسلمه رحمه الله نه الله جل شانه هغه ته یې لوړ مقام ورکړو، محدث، او دنیا کښې د هغه خیر خور شو. په دې علاقه باندې هم په احادیثو کښې د هغه روایتونه شته.

خلاصہ

عرض مې دا کولو چې مونږ ته، چې کله یو مومن مسلمان غافل نه وي، غفلت په وخت کښې راغلی راغلی، غفلت راغلی، ځوانۍ کښې ګناه کښې سړی لګیا دی، دې ته یې سوچ او فکر نشته، فوراً د غفلت نه راووځه. که ستا عمر ډوب تذکر، د الله یاد، الله جل شانه نه یې وېره، د ګناهونو وېره، الله یاد، بیا په زړه کښې پیدا شو. چې ورته وویل چې ما خو سوه قتلونه کړي دي. نو دې چې ته وویل، چې یاره، په دې کلي کښې نه، په هغه بل کلي کښې یو عالم سړی دی، هغه ته ورشه. هغه لاړو، هغه عابد ورته وویل چې ته، له کوم سړي توبه نه قبلیږي، زه ورپسې. هغه د لاره په منځ کښې په مرګ ونیول شو. د ده نیت او پوره وه، خو ورته وویل چې هو، د الله جل شانه مهربان دی، هغه به دې معاف کړي، خو ته له دې کلي ووځه. او چې محمد کلي ته لاړ شه. هغه چې د پښې په تیریدو د لار په منځ کښې په مرګ اخته شو، د ده نیت او پوره شو، خو ورسېدلو نه مخکې مړ شو. خو بعضې روایت کښې دي چې نیکانو فرښتو او بدو ارواحو فرښتو او بعضې د عذاب ګناهګارو کافرو، په دې باندې جګړه شوه. یو فرښتې وویل چې مونږ دا نیکو ته وړو، د توبې په اراده باندې روان شوی وو، مونږ یې روح قبض کوو، او بل قسم فرښتو وویل چې نه، تر څو پورې لا توبه نه وه وتلې، ګناهګاره دی، مونږ یې روح قبض کوو، د روح عذاب ته یې وړو. بیا د حدیث مبارک کښې راځي چې الله جل شانه هغه ځای ته د ځمکې حکم وکړو چې ته را لنډ شه. نو یو دېرش منه ګناهګاره نه هغه کلي ته نږدې شو، چې کوم کلي ته دی روان وو. الله جل شانه وویل، چې دې اخلاص، په دې توبه، په دې تذکر، د الله جل شانه په دې یاد باندې، او د توبې د سوچ ارادې کله کله په زړه کښې، چې خپل توبه یې شروع کړه، چې د هغه چن او طریقه، هغه چې په لاس باندې توبه وباسي، د هغه دعا وکړي، د هغه کس ته لاړ نه وو، خو سړی د الله جل شانه رسول الله صلی الله علیه وسلم فرمایلي، هغه ته یې ټول ګناهونه معاف کړل.

عرض مې دا کولو چې نن سبا شیطان دا کوشش کوي، چې د مسلمان په زړه باندې هم دغه د حیا او د شرم پردې راوغورځوي، پردې، دا د مال د محبت پرده، دا د خوراک پرده، دا د لباس او پوشاک پرده، دا د عهدي او منصب پرده، دا د سیاست پرده وي، دا د مارکیټینګ پرده وي، دا د نوکري پرده وي، دا د تجارت پرده وي، مختلف قسم د غفلت پردې، هغه په زړونو باندې پرته وي. لیکن مومن مسلمان د هغه غفلت کښې وخت نه تیروي. که چرې د ده وخت په دې غفلت کښې تیر شو، الله جل شانه یې مونږ ته معاف کړي. د الله قهر او غضب په دې غفلت کښې راغلی. د دنیا د سړي ګوري. او بیا چې په مینو هغه د هدایت د لارې نه بې باسه. یاد د الله جل شانه له عذاب ته بیا نو دنیا کښې هم، نو د الله جل شانه یاد، زړه کړه. د الله نوم اخستل پکار دي. د الله جل شانه نوم کښې دومره زور دی، چې عبادت، اعلی عبادت جوړیږي، چې کله په هغې کښې د الله نوم وي. اوس د جمعې مونځ دی، ان شاءالله بل ساعت کښې، د الله جل شانه مونږ ته توفیق راکړي، دا مونږ به د جمعې په مبارک د لمانځه وخت کښې نیت وتړو، نويت أن أصلي صلاة الجمعة لله تعالى خلف هذا الإمام. د الله لپاره لمونځ کوم، الله لپاره. که چېرې دا نیت نه کړي، دا نوم د الله پکښې شامل نه کړي، نو یو سړی نیت د لمانځه ونکړي، نو هغه لمونځ لمونځ نه دی. د الله جل شانه نوم سره دا عبادت شي.

اہم نکات

مسلمان باید د الله تعالی څخه غافل نه وي، ځکه چې د الله عذاب اکثره د غفلت په حالت کښې راځي.

کله چې شیطان یو متقي او پرهیزګاره بنده ته وسوسه واچوي، هغه سمدستي الله تعالی یادوي او د ګناه څخه ځان ژغوري.

د الله تعالی ذکر او یاد د غفلت او د شیطان د وسوسو څخه د خلاصون تر ټولو غوره لاره ده.

د توبې دروازه هر وخت خلاصه ده؛ که یو څوک په رښتیني زړه توبه وباسي، الله تعالی یې ټول ګناهونه معاف کوي، که څه هم ډیر لوی وي.

ښه او نیک ماحول د انسان په اصلاح او د ګناهونو څخه په لرې ساتلو کښې ډیر مهم رول لوبوي.

کله چې په انسان کښې شرم او حیا ختم شي، نو هغه په ښکاره توګه هر ډول ګناه ته زړه ښه کوي.

د الله تعالی نوم دومره برکت لري چې هر عبادت او نیک عمل د هغه په نوم سره پیل کول د قبولیت لامل ګرځي.
